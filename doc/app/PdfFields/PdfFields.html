def build(claim)      
  @claim = claim      
  @insurance_billing = InsuranceBilling.find(@claim)
  @insurance_session = InsuranceSession.find(@insurance_billing.insurance_session_id)
  @patient = @insurance_session.patient    
  
  # if there is only one claim or the claim is the first one, then use it as primary
  # if more than one claim, the currnet claim is secondary and the first claim is primary      
  if @insurance_session.insurance_billings.size == 1 || @insurance_session.insurance_billings.first.id == @insurance_billing.id
    @subscriber = @insurance_billing.subscriber
    @insurance_company = @subscriber.insurance_company
    get_header(@insurance_company)
  else
    @second_subscriber = @insurance_billing.subscriber
    @second_insurance_company = @second_subscriber.insurance_company
    @subscriber = @insurance_session.insurance_billings.first.subscriber
    @insurance_company = @subscriber.insurance_company
    get_header(@second_insurance_company)
    secondary_insurance     #second_subscriber && second_insurance_company
  end
  get_patient_info        #patient variable
  get_insured_info        #subscriber && insurance_company           
  get_patient_signatures  #patient && subscriber

  get_default_history  #set box 10 cbk's to no
  if @insurance_session.patient_injury
    @patient_injury = @insurance_session.patient_injury
    get_history_info        #patient_injury  
  end
  
  if @insurance_session.managed_care
    @managed_care = @insurance_session.managed_care
    get_managed_care
  end
  get_refering_provider   #uses patient
  
  @idiagnostics = @insurance_billing.idiagnostics
  @iprocedures = @insurance_billing.iprocedures
  @rendering_provider = @insurance_session.provider      
  if @insurance_session.group?
    @group = @insurance_session.group
    @provider_insurance = @group.provider_insurances.find_by_insurance_company_id(@insurance_company)
  else
    @provider_insurance = @rendering_provider.provider_insurances.find_by_insurance_company_id(@insurance_company)
  end
  get_claim_info          #insurance_session & insurance_billing  
  @service_office = @insurance_session.office       
  @billing_office = @insurance_session.billing_office   
  # in case the billign office is blank, then use the session office
  @billing_office = @insurance_session.office if @billing_office.blank?
  get_provider_info       #provider & group
  return @fields
end

def get_claim_info
  #set the diagnostic codes
  @idiagnostics.each_with_index do |diag, index|
    if index <= 5 #1500 form has a limit of 4 diagnostic codes; the 5th code can be squeezed onto the form
      if diag.dsm_code
        @fields["box21_txt" + (index+1).to_s] = diag.dsm_code            
      elsif diag.icd9_code
        @fields["box21_txt" + (index+1).to_s] = diag.icd9_code
      elsif diag.dsm4_code
        @fields["box21_txt" + (index+1).to_s] = diag.dsm4_code
      elsif diag.dsm5_code
        @fields["box21_txt" + (index+1).to_s] = diag.dsm5_code
      elsif diag.icd10_code
        @fields["box21_txt" + (index+1).to_s] = diag.icd10_code
      end
    end # end if index...
  end  # end of do each...
  #set the value for the diagnostic pointer
  case @idiagnostics.size
    when 1
      @diag_ptr = '1'
    when 2
      @diag_ptr = '1-2'
    when 3
      @diag_ptr = '1-3'
    when 4
      @diag_ptr = '1-4'
    else
      @diag_ptr = '1-5'
  end
    
  
  #set the procedure codes
  @iprocedures.each_with_index do |proc, index|
    i = (index + 1).to_s        
    #date of service start
    @fields["box24a" + i + "_txt1"] = @insurance_session.dos.month
    @fields["box24a" + i + "_txt2"] = @insurance_session.dos.day
    @fields["box24a" + i + "_txt3"] = @insurance_session.dos.year
    #date of service stop
    @fields["box24a" + i + "_txt4"] = @insurance_session.dos.month
    @fields["box24a" + i + "_txt5"] = @insurance_session.dos.day
    @fields["box24a" + i + "_txt6"] = @insurance_session.dos.year
    #place of service
    @fields["box24b" + i + "_txt1"] = @insurance_session.pos_code
    #cpt and modifiers
    @fields["box24d" + i + "_txt1"] = proc.cpt_code
    
    #modifiers may have been entered skipping spaces
    modindex = 2
    [proc.modifier1, proc.modifier2, proc.modifier3, proc.modifier4].each do |mod|
      if !mod.blank?            
        @fields["box24d" + i + "_txt" + modindex.to_s] = mod
        modindex += 1
      end  
    end
    
    #deal with diagnostic pointer
    @fields["box24e" + i + "_txt1"] = @diag_ptr
    
    # enter the charge from the rate_id or rate_override 
    @fields["box24f" + i + "_txt1"] = !proc.rate_override.blank? ? sprintf("%.2f", proc.rate_override) : sprintf("%.2f", proc.rate.rate) 
    # fillin the rendering provider NPI
    @fields["box24j" + i + "_txt2"] = @rendering_provider.npi
  end  #end of iprocedures.each ....     
  #fill in the remaining claim info
  @fields["box28_txt1"] = sprintf("%.2f", @insurance_session.charges_for_service)
  @fields["box29_txt1"] = sprintf("%.2f", @insurance_session.patient_copay)
  @fields["box30_txt1"] = sprintf("%.2f", @insurance_session.balance_owed)
end

def get_header(ins_company)
  header = ins_company.name + "\n"
  header += !ins_company.insurance_co_id.blank? ? (ins_company.insurance_co_id + "\n") : ""
  header += ins_company.address1 + "\n"
  header += !ins_company.address2.blank? ? (ins_company.address2 + "\n") : ""
  header += ins_company.city + ", " + ins_company.state + "  " + ins_company.zip 
  @fields["header_txt1"] = header
end

def get_history_info
  case @patient_injury.accident_type
  when "Employment"
    @fields["box10a_cbk1"] = "X"
    @fields["box10a_cbk2"] = ""
  when "Auto"
    @fields["box10b_cbk1"] = "X"
    @fields["box10b_cbk2"] = ""
    @fields["box10b_txt1"] = @patient_injury.accident_state 
  when "Other"
    @fields["box10c_cbk1"] = "X"
    @fields["box10c_cbk2"] = ""
  end
  
  #box 16 - unable to work
  if @patient_injury.unable_to_work_start
    @fields["box16_txt1"] = @patient_injury.unable_to_work_start.month
    @fields["box16_txt2"] = @patient_injury.unable_to_work_start.day
    @fields["box16_txt3"] = @patient_injury.unable_to_work_start.year
  end
  if @patient_injury.unable_to_work_stop
    @fields["box16_txt4"] = @patient_injury.unable_to_work_stop.month
    @fields["box16_txt5"] = @patient_injury.unable_to_work_stop.day
    @fields["box16_txt6"] = @patient_injury.unable_to_work_stop.year
  end
  
  #box 18 - hospitalization
  if @patient_injury.hospitalization_start
    @fields["box18_txt1"] = @patient_injury.hospitalization_start.month
    @fields["box18_txt2"] = @patient_injury.hospitalization_start.day
    @fields["box18_txt3"] = @patient_injury.hospitalization_start.year
  end
  if @patient_injury.hospitalization_stop
    @fields["box18_txt4"] = @patient_injury.hospitalization_stop.month
    @fields["box18_txt5"] = @patient_injury.hospitalization_stop.day
    @fields["box18_txt6"] = @patient_injury.hospitalization_stop.year
  end
  @fields["box14_txt1"] = @patient_injury.start_illness.month
  @fields["box14_txt2"] = @patient_injury.start_illness.day
  @fields["box14_txt3"] = @patient_injury.start_illness.year
end

def get_insured_info
  # box 1 maps to subscriber, type of insurance
  case @subscriber.type_insurance
  when "Medicare"
    @fields["box1_cbk1"] = "X"
  when "Medicaid"
    @fields["box1_cbk2"] = "X"
  when "Tricare Champus"
    @fields["box1_cbk3"] = "X"
  when "ChampVA"
    @fields["box1_cbk4"] = "X"
  when "Group"
    @fields["box1_cbk5"] = "X"
  else #"other"
    @fields["box1_cbk7"] = "X"
  end

  # box 1a maps to subscriber, policy number
  @fields["box1a_txt1"] = @subscriber.ins_policy
  # box 4 maps to clinet_insured subscribers name
  @fields["box4_txt1"] = @subscriber.subscriber_name

  # box 6 maps to subscriber, type_patient
  case @subscriber.type_patient
  when "Self"
    @fields["box6_cbk1"] = "X"
  when "Spouse"
    @fields["box6_cbk2"] = "X"
  when "Child"
    @fields["box6_cbk3"] = "X"
  else #other
    @fields["box6_cbk4"] = "X" 
  end

  # box 7 maps to subscriber, subscriber
  if @subscriber.subscriber_address2
    @fields["box7_txt1"] = @subscriber.subscriber_address1 + ", " + @subscriber.subscriber_address2
  else
    @fields["box7_txt1"] = @subscriber.subscriber_address1
  end
  @fields["box7_txt2"] = @subscriber.subscriber_city
  @fields["box7_txt3"] = @subscriber.subscriber_state
  @fields["box7_txt4"] = @subscriber.subscriber_zip
  @fields["box7_txt5"] = ""
  # box 11 is the group number in subscriber
  @fields["box11_txt1"] = @subscriber.ins_group
  # box 11 maps to subscriber dob and gender
  @fields["box11a_txt1"] = @subscriber.subscriber_dob.month
  @fields["box11a_txt2"] = @subscriber.subscriber_dob.day
  @fields["box11a_txt3"] = @subscriber.subscriber_dob.year
  if @subscriber.subscriber_gender == Patient::GENDER[0] #male
    @fields["box11a_cbk1"] = "X"
  else
    @fields["box11a_cbk2"] = "X"  
  end      
  # box 11b - subscriber employer
  @fields["box11b_txt1"] = @subscriber.employer_name
  # box 11c - subscriber plan name
  @fields["box11c_txt1"] = @insurance_company.name
  
end

def get_patient_info      
  # box 2 maps to patient name
  @fields["box2_txt1"] = @patient.patient_name
  # box 3 maps to patient dob and gender
  @fields["box3_txt1"] = @patient.dob.month
  @fields["box3_txt2"] = @patient.dob.day
  @fields["box3_txt3"] = @patient.dob.year
  if @patient.gender == Patient::GENDER[0] #male
    @fields["box3_cbk1"] = "X"
  else
    @fields["box3_cbk2"] = "X"  
  end      
  
  # box 5 maps to patient
  if @patient.address2
    @fields["box5_txt1"] = @patient.address1 + ", " + @patient.address2
  else
    @fields["box5_txt1"] = @patient.address1
  end
  @fields["box5_txt2"] = @patient.city
  @fields["box5_txt3"] = @patient.state
  @fields["box5_txt4"] = @patient.zip
  @fields["box5_txt5"] = @patient.home_phone
  
  
  # box 8 maps to patient, patient status and relationship status
  case @patient.relationship_status
  when Patient::RELATIONSHIP[0]
    @fields["box8_cbk1"] = "X"
  when Patient::RELATIONSHIP[1]
    @fields["box8_cbk2"] = "X"
  else # other
    @fields["box8_cbk3"] = "X"
  end
  
  case @patient.patient_status
  when Patient::PATIENT_STATUS[0]
    @fields["box8_cbk4"] = "X"
  when Patient::PATIENT_STATUS[1]
    @fields["box8_cbk5"] = "X"
  else #part time student
    @fields["box8_cbk6"] = "X"
  end
end

def get_provider_info
  if @group
    #box 25 ein number
    if @provider_insurance && !@provider_insurance.ein_suffix.blank?      
      @fields["box25_txt1"] = @group.ein_number + @provider_insurance.ein_suffix
    else
      @fields["box25_txt1"] = @group.ein_number
    end
          
    @fields["box25_cbk2"] = "X"          
  else
    #box 25 ssn or ein number
    if !@rendering_provider.ein_number.blank?
      if @provider_insurance && !@provider_insurance.ein_suffix.blank?
        @fields["box25_txt1"] = @rendering_provider.ein_number + @provider_insurance.ein_suffix
      else
        @fields["box25_txt1"] = @rendering_provider.ein_number  
      end
      
      @fields["box25_cbk2"] = "X"
    else
      @fields["box25_txt1"] = @rendering_provider.ssn_number
      @fields["box25_cbk1"] = "X"
    end        
  end                
  
  #box 26 - patient account number, enter the unique claim number
  @fields["box26_txt1"] = @insurance_billing.claim_number
  
  #box 27 - accepts assignment, maps to patient.accepts_assignement
  if @patient.accept_assignment
    @fields["box27_cbk1"] = "X"
  else
    @fields["box27_cbk2"] = "X"
  end 
  #box 31 signature on file maps to provider
  if @rendering_provider.signature_on_file
    @fields["box31_txt1"] = @rendering_provider.provider_name
    @fields["box31_txt2"] = DateTime.now.strftime('%m/%d/%Y')
  end
  #box 32, 33 office & provider info      
  #collect & format the service facility office
  service = !@service_office.office_name.blank? ? @service_office.office_name : @rendering_provider.provider_name
  service += "\n" + @service_office.address1      
  service += !@service_office.address2.blank? ? (", " + @service_office.address2) : ""
  service += "\n" + @service_office.city + ", " + @service_office.state + "  " + @service_office.zip        
  #collect & format the billing office
  billing = !@billing_office.office_name.blank? ? @billing_office.office_name : @rendering_provider.provider_name
  billing += "\n" + @billing_office.address1      
  billing += !@billing_office.address2.blank? ? (", " + @billing_office.address2) : ""
  billing += "\n" + @billing_office.city + ", " + @billing_office.state + "  " + @billing_office.zip        
  
  
  
  @fields["box32_txt1"] = service
  @fields["box32a_txt1"] = @rendering_provider.npi
  @fields["box32b_txt1"] = @rendering_provider.npi
  @fields["box33_txt1"] = @rendering_provider.office_phone
  @fields["box33_txt2"] = billing
  @fields["box33a_txt1"] = @rendering_provider.npi
end

def render
  @xml.instruct!
  @xfdf.print @xml.xfdf("xmlns" => "http://ns.adobe.com/xfdf/", "xml:space" => "preserve") {
    @xml.fields {
      @fields.each do |field, value|
        @xml.field(:name => field) {
          if value.is_a? Array
            value.each {|item| @xml.value(item.to_s) }
          else
            @xml.value(value.to_s)
          end
      }
    end
    }
  }
  @xfdf.close      
  return @xfdf
end
