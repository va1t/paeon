def ajax_detail
  @balance_bill = BalanceBill.find(params[:balance_bill_id])
  @details = []
  #if there are added records, then make sure to include them in the returned form
  params[:balance_bill][:balance_bill_details_attributes].each { |key, value| @details.push @balance_bill.balance_bill_details.build(value) } if params[:balance_bill]
  @count = @balance_bill.balance_bill_details.count
  @details.push @balance_bill.balance_bill_details.build()

  respond_to do |format|    
    format.js {render :layout => false }                   
  end
end

def new
  @insurance_session = InsuranceSession.find(params[:insurance_session_id])
  # verify that all claims and balance bills are closed before opening a new claim
  if @insurance_session.verify_session_state      
    @balance_bill = @insurance_session.balance_bills.new(:created_user => current_user.login_name, :patient_id => @insurance_session.patient_id, 
                                                         :group_id => @insurance_session.group_id, :provider_id => @insurance_session.provider_id,
                                                         :status => BalanceBillFlow::INITIATE)
    # the balance_due is set from the session's balance_owed.  if null balance_owed, then set to 0.
    # on every balance bill update and/or ins billing update, the session charges are recalculated, so the balance_owed is accurate
    @balance_bill.total_invoiced_amount = @insurance_session.balance_owed.blank? ? 0 : @insurance_session.balance_owed
    if @balance_bill.save!
      #create the first detail record for the balance bill.
      @balance_detail = @balance_bill.balance_bill_details.new(:created_user => current_user.login_name, :amount => @insurance_session.balance_owed, :quantity => 1, 
                                                             :description => "Balance owed for date of service: #{@insurance_session.dos.strftime("%m/%d/%Y")}")
      @balance_detail.save!
      message = 'Balance bill was successfully created.'
    else
      message = ""            
      @balance_bill.errors.full_messages.each {|msg| message += msg + "\n"} if @balance_bill.errors.any?                                
    end
  else
    # cant create a new blance bill because either an existing balance bil is open or a claim is open
    message = ""            
    @insurance_session.errors.full_messages.each {|msg| message += msg + "\n"} if @insurance_session.errors.any?                      
  end      
  respond_to do |format|
    # redirect to the index method to display 
    format.html { redirect_to insurance_session_balance_bills_path, notice: message }
    format.json { head :no_content }
  end
end

def update
  @insurance_session = InsuranceSession.find(params[:insurance_session_id])
  @balance_bill = @insurance_session.balance_bills.find(params[:id])
  @balance_bill.updated_user = current_user.login_name
  
  #sum up all the detail records
  @balance_bill.total_invoiced_amount = 0
  if params[:balance_bill]
    #update the fields for the details record   
    @details = params[:balance_bill][:balance_bill_details_attributes]    
    if @details
      @details.each do |x, d|
        d[:created_user] = current_user.login_name if d[:created_user].blank?
        d[:updated_user] = current_user.login_name if d[:updated_user].blank?
        @balance_bill.total_invoiced_amount += d[:amount].to_f if !d[:amount].blank?
      end
    end
  end    

  respond_to do |format|
    if @balance_bill.update_attributes(params[:balance_bill])
      format.html { 
        if params[:commit] == "Update Balance Bill"
          redirect_to insurance_session_balance_bills_path, notice: 'Balance Bill was successfully updated.'
        else
          redirect_to insurance_sessions_path, notice: 'Balance Bill was successfully updated.'
        end }        
      format.json { head :no_content }
    else
      notice = 'Error saving.'
      @balance_bill.errors.full_messages.each {|msg| notice += " " + msg } if @balance_bill.errors.any?
      format.html { redirect_to insurance_session_balance_bills_path, notice: notice }
      format.json { render json: @balance_bill.errors, status: :unprocessable_entity }
    end
  end
end
