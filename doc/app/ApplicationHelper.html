def display_group_link
  regexp = /^\/groups\/\d+$/
  if regexp.match request.fullpath
    link = link_to session[:group_name], edit_group_path(session[:group])
  else
    link = link_to session[:group_name], group_path(session[:group])
  end
  return ("Group: " + link).html_safe
end
